\begin{align*}
\frac{1}{2}\,\lVert \vec{(A-B)}\times\vec{(A-C)}\rVert
\end{align*}


\begin{align}
   \vec{A}-\vec{B} = \myvec{-1 \\ 0} - \myvec{1 \\ 3} = \myvec{-2 \\ -3}
\end{align}

\begin{align}
   \vec{A}-\vec{C} = \myvec{-1 \\ 0} - \myvec{3 \\ 2} = \myvec{-4 \\ -2}
\end{align}


\begin{align}
\frac{1}{2}\,\lVert \vec{(A-B)}\times\vec{(A-C)}\rVert = 4
\end{align}

\bigskip
