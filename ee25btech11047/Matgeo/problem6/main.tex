\textbf{Question:}  
Find the equation of the line passing through the points  
\[
\vec{A} = \myvec{1 \\ 2}, 
\quad 
\vec{B} = \myvec{3 \\ 6}.
\]

\textbf{Solution:}  

\begin{align}
\myvec{\vec{A}&\vec{B}}^T\vec{n} &= \myvec{1\\1}\\
\myvec{1 & 2 \\ 3 & 6}\vec{n} &= \myvec{1 \\ 1}\\ 
\myvec{1 & 2 & 1 \\ 3 & 6 & 1} & 
\xrightarrow{R_2 - 3R_1}\;\;
\myvec{1 & 2 & 1 \\ 0 & 0 & -2} \\
0=-2
\end{align}

\xrightarrow{R_2 - 3R_1}\;\;
\myvec{1 & 2 & 0 \\ 0 & 0 & 0}

\vec{n} &= \myvec{-2 \\ 1}
\end{align}
